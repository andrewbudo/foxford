Можно честно промоделировать процесс, так как задач до двух миллиардов, а нам потребуется порядка корня из этого числа дней.

n = int(input())

k = int(input())

summ = 0

for i in range(10 ** 10):

    summ += k

    k += 1

    if summ >= n:

        print(i + 1)

        break
